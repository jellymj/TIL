  > a = 0.1 * 3
  >
  > b = 0.3

  ```python
  a = 0.1 *3
  b = 0.3
  print(bool(abs(a-b) <= 1e-10))
  
  
  import sys
  print(bool(abs(a-b) <= sys.float_info.epsilon))
  
  import math
  print(bool(math.isclose(a,b)))
  ```
